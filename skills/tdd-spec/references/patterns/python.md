# Python Test Patterns

Test patterns for Python projects using pytest.

## Framework Detection

```bash
# Check for pytest configuration
test -f "pytest.ini" || test -f "pyproject.toml" && grep pytest pyproject.toml
```

## Imports

```python
import pytest
from unittest.mock import Mock, patch, MagicMock

# Component under test (will fail until implemented)
from {layer}.{component_module} import ComponentName

# Types
from types.{type_module} import InputType, OutputType, ErrorType
```

## Test Structure

```python
"""Tests for ComponentName.

Generated by /opensdd:tdd-spec from spec.yaml
Component: ComponentName
"""

import pytest
from unittest.mock import Mock, patch


class TestComponentName:
    """Test suite for ComponentName."""

    @pytest.fixture
    def component(self):
        """Create component instance for testing."""
        return ComponentName()

    # Happy path tests
    class TestFunctionName:
        """Tests for function_name method."""

        def test_describes_expected_behavior(self, component):
            """Test that function_name does expected thing."""
            # Arrange
            input_data = InputType(field="value")

            # Act
            result = component.function_name(input_data)

            # Assert
            assert isinstance(result, OutputType)
            assert result.field == expected_value

        def test_returns_error_when_condition(self, component):
            """Test error case for function_name."""
            # Arrange
            input_data = InputType(field="invalid")

            # Act
            result = component.function_name(input_data)

            # Assert
            assert isinstance(result, ErrorType)
            assert result.code == "EXPECTED_ERROR_CODE"

        def test_handles_none_input(self, component):
            """Test edge case: None input."""
            with pytest.raises(ValueError):
                component.function_name(None)

        def test_handles_empty_input(self, component):
            """Test edge case: empty input."""
            input_data = InputType(field="")
            result = component.function_name(input_data)
            assert isinstance(result, ErrorType)
```

## Assertions

```python
# Equality
assert value == expected
assert value != unexpected
assert value is None
assert value is not None

# Type checking
assert isinstance(value, ExpectedType)
assert type(value) is ExactType

# Truthiness
assert value
assert not value

# Comparisons
assert value > 0
assert value >= min_val
assert value < max_val
assert min_val <= value <= max_val

# Collections
assert item in collection
assert len(collection) == expected_length
assert collection == expected_list

# Exceptions
with pytest.raises(ExpectedError):
    function_that_raises()

with pytest.raises(ExpectedError) as exc_info:
    function_that_raises()
assert "expected message" in str(exc_info.value)

# Approximate equality (floats)
assert value == pytest.approx(expected, rel=1e-3)
```

## Fixtures

```python
@pytest.fixture
def component():
    """Create fresh component for each test."""
    return ComponentName()

@pytest.fixture
def mock_dependency():
    """Create mock for external dependency."""
    mock = Mock(spec=DependencyClass)
    mock.method.return_value = expected_value
    return mock

@pytest.fixture
def component_with_mocks(mock_dependency):
    """Create component with mocked dependencies."""
    return ComponentName(dependency=mock_dependency)

# Fixture with cleanup
@pytest.fixture
def resource():
    """Fixture with setup and teardown."""
    resource = acquire_resource()
    yield resource
    release_resource(resource)
```

## Mocking

```python
from unittest.mock import Mock, patch, MagicMock

# Mock an object
mock_obj = Mock()
mock_obj.method.return_value = expected_value
mock_obj.method.side_effect = Exception("error")

# Patch a module function
@patch('module.path.function_name')
def test_with_patch(mock_function):
    mock_function.return_value = expected_value
    result = code_under_test()
    mock_function.assert_called_once_with(expected_arg)

# Patch as context manager
def test_with_context_patch():
    with patch('module.path.function_name') as mock_fn:
        mock_fn.return_value = expected_value
        result = code_under_test()

# Verify calls
mock_obj.method.assert_called()
mock_obj.method.assert_called_once()
mock_obj.method.assert_called_with(arg1, arg2)
mock_obj.method.assert_not_called()
assert mock_obj.method.call_count == 3
```

## Event Testing

```python
class TestEvents:
    """Tests for event emissions."""

    def test_emits_event_on_condition(self, component):
        """Test that EventName is emitted when condition met."""
        # Arrange
        event_handler = Mock()
        component.on("EventName", event_handler)

        # Act
        component.trigger_action()

        # Assert
        event_handler.assert_called_once()
        call_args = event_handler.call_args[0][0]
        assert "expected_field" in call_args
```

## Async Testing

```python
import pytest

@pytest.mark.asyncio
async def test_async_function():
    """Test async function."""
    result = await async_function(input_data)
    assert result == expected

@pytest.mark.asyncio
async def test_async_raises():
    """Test async function raises error."""
    with pytest.raises(ExpectedError):
        await async_function(bad_input)
```

## Parametrized Tests

```python
@pytest.mark.parametrize("input_val,expected", [
    ("valid", OutputType(success=True)),
    ("invalid", ErrorType(code="INVALID")),
    ("", ErrorType(code="EMPTY")),
    (None, pytest.raises(ValueError)),
])
def test_function_with_various_inputs(component, input_val, expected):
    """Test function_name with various inputs."""
    if isinstance(expected, type(pytest.raises(Exception))):
        with expected:
            component.function_name(input_val)
    else:
        result = component.function_name(input_val)
        assert result == expected
```

## Test Data Factories

```python
from dataclasses import dataclass
from typing import Optional

def create_credentials(
    email: str = "test@example.com",
    password: str = "TestPassword123",
    **overrides
) -> Credentials:
    """Factory for test credentials."""
    return Credentials(
        email=email,
        password=password,
        **overrides
    )

# Usage
def test_authenticates_valid_credentials(component):
    credentials = create_credentials()
    result = component.login(credentials)
    assert isinstance(result, AuthResult)

def test_rejects_invalid_email(component):
    credentials = create_credentials(email="invalid")
    result = component.login(credentials)
    assert isinstance(result, AuthError)
```
